62 /* File Info : -----------------------------------------------------------------
63  User NOTES
64 1. How to use this driver:
65 --------------------------
66  - This driver does not need a specific component driver for the micro SD device
67  to be included with.
68 
69 2. Driver description:
70 ---------------------
71  + Initialization steps:
72  o Initialize the micro SD card using the BSP_SD_Init() function.
73  o Checking the SD card presence is not managed because SD detection pin is
74  not physically mapped on the Adafruit shield.
75  o The function BSP_SD_GetCardInfo() is used to get the micro SD card information
76  which is stored in the structure "SD_CardInfo".
77 
78  + Micro SD card operations
79  o The micro SD card can be accessed with read/write block(s) operations once
80  it is ready for access. The access can be performed in polling
81  mode by calling the functions BSP_SD_ReadBlocks()/BSP_SD_WriteBlocks()
82 
83  o The SD erase block(s) is performed using the function BSP_SD_Erase() with
84  specifying the number of blocks to erase.
85  o The SD runtime status is returned when calling the function BSP_SD_GetStatus().

338 uint8_t BSP_SD_ReadBlocks(uint32_t* pData, uint32_t ReadAddr, uint16_t BlockSize, uint32_t NumberOfBlocks)
339 {
340  uint32_t offset = 0;
341  uint8_t retr = BSP_SD_ERROR;
342  uint8_t *ptr = NULL;
343  SD_CmdAnswer_typedef response;
344 
345  /* Send CMD16 (SD_CMD_SET_BLOCKLEN) to set the size of the block and
346  Check if the SD acknowledged the set block length command: R1 response (0x00: no errors) */
347  response = SD_SendCmd(SD_CMD_SET_BLOCKLEN, BlockSize, 0xFF, SD_ANSWER_R1_EXPECTED);
348  SD_IO_CSState(1);
349  SD_IO_WriteByte(SD_DUMMY_BYTE);
350  if ( response.r1 != SD_R1_NO_ERROR)
351  {
352  goto error;
353  }
354 
355  ptr = malloc(sizeof(uint8_t)*BlockSize);
356  if( ptr == NULL )
357  {
358  goto error;
359  }
360  memset(ptr, SD_DUMMY_BYTE, sizeof(uint8_t)*BlockSize);
361 
362  /* Data transfer */
363  while (NumberOfBlocks--)
364  {
365  /* Send CMD17 (SD_CMD_READ_SINGLE_BLOCK) to read one block */
366  /* Check if the SD acknowledged the read block command: R1 response (0x00: no errors) */
367  response = SD_SendCmd(SD_CMD_READ_SINGLE_BLOCK, (ReadAddr + offset)/(flag_SDHC == 1 ?BlockSize: 1), 0xFF, SD_ANSWER_R1_EXPECTED);
368  if ( response.r1 != SD_R1_NO_ERROR)
369  {
370  goto error;
371  }
372 
373  /* Now look for the data token to signify the start of the data */
374  if (SD_WaitData(SD_TOKEN_START_DATA_SINGLE_BLOCK_READ) == BSP_SD_OK)
375  {
376  /* Read the SD block data : read NumByteToRead data */
377  SD_IO_WriteReadData(ptr, (uint8_t*)pData + offset, BlockSize);
378 
379  /* Set next read address*/
380  offset += BlockSize;
381  /* get CRC bytes (not really needed by us, but required by SD) */
382  SD_IO_WriteByte(SD_DUMMY_BYTE);
383  SD_IO_WriteByte(SD_DUMMY_BYTE);
384  }
385  else
386  {
387  goto error;
388  }
389 
390  /* End the command data read cycle */
391  SD_IO_CSState(1);
392  SD_IO_WriteByte(SD_DUMMY_BYTE);
393  }
394 
395  retr = BSP_SD_OK;
396 
397 error :
398  /* Send dummy byte: 8 Clock pulses of delay */
399  SD_IO_CSState(1);
400  SD_IO_WriteByte(SD_DUMMY_BYTE);
401  if(ptr != NULL) free(ptr);
402 
403  /* Return the reponse */
404  return retr;
405 }

555 uint8_t SD_GetCSDRegister(SD_CSD* Csd)
556 {
557  uint16_t counter = 0;
558  uint8_t CSD_Tab[16];
559  uint8_t retr = BSP_SD_ERROR;
560  SD_CmdAnswer_typedef response;
561 
562  /* Send CMD9 (CSD register) or CMD10(CSD register) and Wait for response in the R1 format (0x00 is no errors) */
563  response = SD_SendCmd(SD_CMD_SEND_CSD, 0, 0xFF, SD_ANSWER_R1_EXPECTED);
564  if(response.r1 == SD_R1_NO_ERROR)
565  {
566  if (SD_WaitData(SD_TOKEN_START_DATA_SINGLE_BLOCK_READ) == BSP_SD_OK)
567  {
568  for (counter = 0; counter < 16; counter++)
569  {
570  /* Store CSD register value on CSD_Tab */
571  CSD_Tab[counter] = SD_IO_WriteByte(SD_DUMMY_BYTE);
572  }
573 
574  /* Get CRC bytes (not really needed by us, but required by SD) */
575  SD_IO_WriteByte(SD_DUMMY_BYTE);
576  SD_IO_WriteByte(SD_DUMMY_BYTE);
577 
578  /*************************************************************************
579  CSD header decoding
580  *************************************************************************/
581 
582  /* Byte 0 */
583  Csd->CSDStruct = (CSD_Tab[0] & 0xC0) >> 6;
584  Csd->Reserved1 = CSD_Tab[0] & 0x3F;
585 
586  /* Byte 1 */
587  Csd->TAAC = CSD_Tab[1];
588 
589  /* Byte 2 */
590  Csd->NSAC = CSD_Tab[2];
591 
592  /* Byte 3 */
593  Csd->MaxBusClkFrec = CSD_Tab[3];
594 
595  /* Byte 4/5 */
596  Csd->CardComdClasses = (CSD_Tab[4] << 4) | ((CSD_Tab[5] & 0xF0) >> 4);
597  Csd->RdBlockLen = CSD_Tab[5] & 0x0F;
598 
599  /* Byte 6 */
600  Csd->PartBlockRead = (CSD_Tab[6] & 0x80) >> 7;
601  Csd->WrBlockMisalign = (CSD_Tab[6] & 0x40) >> 6;
602  Csd->RdBlockMisalign = (CSD_Tab[6] & 0x20) >> 5;
603  Csd->DSRImpl = (CSD_Tab[6] & 0x10) >> 4;
604 
605  /*************************************************************************
606  CSD v1/v2 decoding
607  *************************************************************************/
608 
609  if(flag_SDHC == 0)
610  {
611  Csd->version.v1.Reserved1 = ((CSD_Tab[6] & 0x0C) >> 2);
612 
613  Csd->version.v1.DeviceSize = ((CSD_Tab[6] & 0x03) << 10)
614  | (CSD_Tab[7] << 2)
615  | ((CSD_Tab[8] & 0xC0) >> 6);
616  Csd->version.v1.MaxRdCurrentVDDMin = (CSD_Tab[8] & 0x38) >> 3;
617  Csd->version.v1.MaxRdCurrentVDDMax = (CSD_Tab[8] & 0x07);
618  Csd->version.v1.MaxWrCurrentVDDMin = (CSD_Tab[9] & 0xE0) >> 5;
619  Csd->version.v1.MaxWrCurrentVDDMax = (CSD_Tab[9] & 0x1C) >> 2;
620  Csd->version.v1.DeviceSizeMul = ((CSD_Tab[9] & 0x03) << 1)
621  |((CSD_Tab[10] & 0x80) >> 7);
622  }
623  else
624  {
625  Csd->version.v2.Reserved1 = ((CSD_Tab[6] & 0x0F) << 2) | ((CSD_Tab[7] & 0xC0) >> 6);
626  Csd->version.v2.DeviceSize= ((CSD_Tab[7] & 0x3F) << 16) | (CSD_Tab[8] << 8) | CSD_Tab[9];
627  Csd->version.v2.Reserved2 = ((CSD_Tab[10] & 0x80) >> 8);
628  }
629 
630  Csd->EraseSingleBlockEnable = (CSD_Tab[10] & 0x40) >> 6;
631  Csd->EraseSectorSize = ((CSD_Tab[10] & 0x3F) << 1)
632  |((CSD_Tab[11] & 0x80) >> 7);
633  Csd->WrProtectGrSize = (CSD_Tab[11] & 0x7F);
634  Csd->WrProtectGrEnable = (CSD_Tab[12] & 0x80) >> 7;
635  Csd->Reserved2 = (CSD_Tab[12] & 0x60) >> 5;
636  Csd->WrSpeedFact = (CSD_Tab[12] & 0x1C) >> 2;
637  Csd->MaxWrBlockLen = ((CSD_Tab[12] & 0x03) << 2)
638  |((CSD_Tab[13] & 0xC0) >> 6);
639  Csd->WriteBlockPartial = (CSD_Tab[13] & 0x20) >> 5;
640  Csd->Reserved3 = (CSD_Tab[13] & 0x1F);
641  Csd->FileFormatGrouop = (CSD_Tab[14] & 0x80) >> 7;
642  Csd->CopyFlag = (CSD_Tab[14] & 0x40) >> 6;
643  Csd->PermWrProtect = (CSD_Tab[14] & 0x20) >> 5;
644  Csd->TempWrProtect = (CSD_Tab[14] & 0x10) >> 4;
645  Csd->FileFormat = (CSD_Tab[14] & 0x0C) >> 2;
646  Csd->Reserved4 = (CSD_Tab[14] & 0x03);
647  Csd->crc = (CSD_Tab[15] & 0xFE) >> 1;
648  Csd->Reserved5 = (CSD_Tab[15] & 0x01);
649 
650  retr = BSP_SD_OK;
651  }
652  }
653 
654  /* Send dummy byte: 8 Clock pulses of delay */
655  SD_IO_CSState(1);
656  SD_IO_WriteByte(SD_DUMMY_BYTE);
657 
658  /* Return the reponse */
659  return retr;
660 }

762 SD_CmdAnswer_typedef SD_SendCmd(uint8_t Cmd, uint32_t Arg, uint8_t Crc, uint8_t Answer)
763 {
764  uint8_t frame[SD_CMD_LENGTH], frameout[SD_CMD_LENGTH];
765  SD_CmdAnswer_typedef retr = {0xFF, 0xFF , 0xFF, 0xFF, 0xFF};
766 
767  /* R1 Lenght = NCS(0)+ 6 Bytes command + NCR(min1 max8) + 1 Bytes answer + NEC(0) = 15bytes */
768  /* R1b identical to R1 + Busy information */
769  /* R2 Lenght = NCS(0)+ 6 Bytes command + NCR(min1 max8) + 2 Bytes answer + NEC(0) = 16bytes */
770 
771  /* Prepare Frame to send */
772  frame[0] = (Cmd | 0x40); /* Construct byte 1 */
773  frame[1] = (uint8_t)(Arg >> 24); /* Construct byte 2 */
774  frame[2] = (uint8_t)(Arg >> 16); /* Construct byte 3 */
775  frame[3] = (uint8_t)(Arg >> 8); /* Construct byte 4 */
776  frame[4] = (uint8_t)(Arg); /* Construct byte 5 */
777  frame[5] = (Crc | 0x01); /* Construct byte 6 */
778 
779  /* Send the command */
780  SD_IO_CSState(0);
781  SD_IO_WriteReadData(frame, frameout, SD_CMD_LENGTH); /* Send the Cmd bytes */
782 
783  switch(Answer)
784  {
785  case SD_ANSWER_R1_EXPECTED :
786  retr.r1 = SD_ReadData();
787  break;
788  case SD_ANSWER_R1B_EXPECTED :
789  retr.r1 = SD_ReadData();
790  retr.r2 = SD_IO_WriteByte(SD_DUMMY_BYTE);
791  /* Set CS High */
792  SD_IO_CSState(1);
793  HAL_Delay(1);
794  /* Set CS Low */
795  SD_IO_CSState(0);
796 
797  /* Wait IO line return 0xFF */
798  while (SD_IO_WriteByte(SD_DUMMY_BYTE) != 0xFF);
799  break;
800  case SD_ANSWER_R2_EXPECTED :
801  retr.r1 = SD_ReadData();
802  retr.r2 = SD_IO_WriteByte(SD_DUMMY_BYTE);
803  break;
804  case SD_ANSWER_R3_EXPECTED :
805  case SD_ANSWER_R7_EXPECTED :
806  retr.r1 = SD_ReadData();
807  retr.r2 = SD_IO_WriteByte(SD_DUMMY_BYTE);
808  retr.r3 = SD_IO_WriteByte(SD_DUMMY_BYTE);
809  retr.r4 = SD_IO_WriteByte(SD_DUMMY_BYTE);
810  retr.r5 = SD_IO_WriteByte(SD_DUMMY_BYTE);
811  break;
812  default :
813  break;
814  }
815  return retr;
816 }

869 uint8_t SD_GoIdleState(void)
870 {
871  SD_CmdAnswer_typedef response;
872  __IO uint8_t counter = 0;
873  /* Send CMD0 (SD_CMD_GO_IDLE_STATE) to put SD in SPI mode and
874  wait for In Idle State Response (R1 Format) equal to 0x01 */
875  do{
876  counter++;
877  response = SD_SendCmd(SD_CMD_GO_IDLE_STATE, 0, 0x95, SD_ANSWER_R1_EXPECTED);
878  SD_IO_CSState(1);
879  SD_IO_WriteByte(SD_DUMMY_BYTE);
880  if(counter >= SD_MAX_TRY)
881  {
882  return BSP_SD_ERROR;
883  }
884  }
885  while(response.r1 != SD_R1_IN_IDLE_STATE);
886 
887 
888  /* Send CMD8 (SD_CMD_SEND_IF_COND) to check the power supply status
889  and wait until response (R7 Format) equal to 0xAA and */
890  response = SD_SendCmd(SD_CMD_SEND_IF_COND, 0x1AA, 0x87, SD_ANSWER_R7_EXPECTED);
891  SD_IO_CSState(1);
892  SD_IO_WriteByte(SD_DUMMY_BYTE);
893  if((response.r1 & SD_R1_ILLEGAL_COMMAND) == SD_R1_ILLEGAL_COMMAND)
894  {
895  /* initialise card V1 */
896  do
897  {
898  /* initialise card V1 */
899  /* Send CMD55 (SD_CMD_APP_CMD) before any ACMD command: R1 response (0x00: no errors) */
900  response = SD_SendCmd(SD_CMD_APP_CMD, 0x00000000, 0xFF, SD_ANSWER_R1_EXPECTED);
901  SD_IO_CSState(1);
902  SD_IO_WriteByte(SD_DUMMY_BYTE);
903 
904  /* Send ACMD41 (SD_CMD_SD_APP_OP_COND) to initialize SDHC or SDXC cards: R1 response (0x00: no errors) */
905  response = SD_SendCmd(SD_CMD_SD_APP_OP_COND, 0x00000000, 0xFF, SD_ANSWER_R1_EXPECTED);
906  SD_IO_CSState(1);
907  SD_IO_WriteByte(SD_DUMMY_BYTE);
908  }
909  while(response.r1 == SD_R1_IN_IDLE_STATE);
910  flag_SDHC = 0;
911  }
912  else if(response.r1 == SD_R1_IN_IDLE_STATE)
913  {
914  /* initialise card V2 */
915  do {
916 
917  /* Send CMD55 (SD_CMD_APP_CMD) before any ACMD command: R1 response (0x00: no errors) */
918  response = SD_SendCmd(SD_CMD_APP_CMD, 0, 0xFF, SD_ANSWER_R1_EXPECTED);
919  SD_IO_CSState(1);
920  SD_IO_WriteByte(SD_DUMMY_BYTE);
921 
922  /* Send ACMD41 (SD_CMD_SD_APP_OP_COND) to initialize SDHC or SDXC cards: R1 response (0x00: no errors) */
923  response = SD_SendCmd(SD_CMD_SD_APP_OP_COND, 0x40000000, 0xFF, SD_ANSWER_R1_EXPECTED);
924  SD_IO_CSState(1);
925  SD_IO_WriteByte(SD_DUMMY_BYTE);
926  }
927  while(response.r1 == SD_R1_IN_IDLE_STATE);
928 
929  if((response.r1 & SD_R1_ILLEGAL_COMMAND) == SD_R1_ILLEGAL_COMMAND)
930  {
931  do {
932  /* Send CMD55 (SD_CMD_APP_CMD) before any ACMD command: R1 response (0x00: no errors) */
933  response = SD_SendCmd(SD_CMD_APP_CMD, 0, 0xFF, SD_ANSWER_R1_EXPECTED);
934  SD_IO_CSState(1);
935  SD_IO_WriteByte(SD_DUMMY_BYTE);
936  if(response.r1 != SD_R1_IN_IDLE_STATE)
937  {
938  return BSP_SD_ERROR;
939  }
940  /* Send ACMD41 (SD_CMD_SD_APP_OP_COND) to initialize SDHC or SDXC cards: R1 response (0x00: no errors) */
941  response = SD_SendCmd(SD_CMD_SD_APP_OP_COND, 0x00000000, 0xFF, SD_ANSWER_R1_EXPECTED);
942  SD_IO_CSState(1);
943  SD_IO_WriteByte(SD_DUMMY_BYTE);
944  }
945  while(response.r1 == SD_R1_IN_IDLE_STATE);
946  }
947 
948  /* Send CMD58 (SD_CMD_READ_OCR) to initialize SDHC or SDXC cards: R3 response (0x00: no errors) */
949  response = SD_SendCmd(SD_CMD_READ_OCR, 0x00000000, 0xFF, SD_ANSWER_R3_EXPECTED);
950  SD_IO_CSState(1);
951  SD_IO_WriteByte(SD_DUMMY_BYTE);
952  if(response.r1 != SD_R1_NO_ERROR)
953  {
954  return BSP_SD_ERROR;
955  }
956  flag_SDHC = (response.r2 & 0x40) >> 6;
957  }
958  else
959  {
960  return BSP_SD_ERROR;
961  }
962 
963  return BSP_SD_OK;
964 }
