170 HAL_StatusTypeDef HAL_RS485Ex_Init(UART_HandleTypeDef *huart, uint32_t Polarity, uint32_t AssertionTime, uint32_t DeassertionTime)
171 {
172  uint32_t temp = 0x0U;
173 
174  /* Check the UART handle allocation */
175  if(huart == NULL)
176  {
177  return HAL_ERROR;
178  }
179  /* Check the Driver Enable UART instance */
180  assert_param(IS_UART_DRIVER_ENABLE_INSTANCE(huart->Instance));
181 
182  /* Check the Driver Enable polarity */
183  assert_param(IS_UART_DE_POLARITY(Polarity));
184 
185  /* Check the Driver Enable assertion time */
186  assert_param(IS_UART_ASSERTIONTIME(AssertionTime));
187 
188  /* Check the Driver Enable deassertion time */
189  assert_param(IS_UART_DEASSERTIONTIME(DeassertionTime));
190 
191  if(huart->gState == HAL_UART_STATE_RESET)
192  {
193  /* Allocate lock resource and initialize it */
194  huart->Lock = HAL_UNLOCKED;
195 
196  /* Init the low level hardware : GPIO, CLOCK, CORTEX */
197  HAL_UART_MspInit(huart);
198  }
199 
200  huart->gState = HAL_UART_STATE_BUSY;
201 
202  /* Disable the Peripheral */
203  __HAL_UART_DISABLE(huart);
204 
205  /* Set the UART Communication parameters */
206  if (UART_SetConfig(huart) == HAL_ERROR)
207  {
208  return HAL_ERROR;
209  }
210 
211  if(huart->AdvancedInit.AdvFeatureInit != UART_ADVFEATURE_NO_INIT)
212  {
213  UART_AdvFeatureConfig(huart);
214  }
215 
216  /* Enable the Driver Enable mode by setting the DEM bit in the CR3 register */
217  SET_BIT(huart->Instance->CR3, USART_CR3_DEM);
218 
219  /* Set the Driver Enable polarity */
220  MODIFY_REG(huart->Instance->CR3, USART_CR3_DEP, Polarity);
221 
222  /* Set the Driver Enable assertion and deassertion times */
223  temp = (AssertionTime << UART_CR1_DEAT_ADDRESS_LSB_POS);
224  temp |= (DeassertionTime << UART_CR1_DEDT_ADDRESS_LSB_POS);
225  MODIFY_REG(huart->Instance->CR1, (USART_CR1_DEDT|USART_CR1_DEAT), temp);
226 
227  /* Enable the Peripheral */
228  __HAL_UART_ENABLE(huart);
229 
230  /* TEACK and/or REACK to check before moving huart->gState and huart->RxState to Ready */
231  return (UART_CheckIdleState(huart));
232 }

307 HAL_StatusTypeDef HAL_UARTEx_StopModeWakeUpSourceConfig(UART_HandleTypeDef *huart, UART_WakeUpTypeDef WakeUpSelection)
308 {
309  HAL_StatusTypeDef status = HAL_OK;

311 
312  /* check the wake-up from stop mode UART instance */
313  assert_param(IS_UART_WAKEUP_FROMSTOP_INSTANCE(huart->Instance));
314  /* check the wake-up selection parameter */
315  assert_param(IS_UART_WAKEUP_SELECTION(WakeUpSelection.WakeUpEvent));
316 
317  /* Process Locked */
318  __HAL_LOCK(huart);
319 
320  huart->gState = HAL_UART_STATE_BUSY;
321 
322  /* Disable the Peripheral */
323  __HAL_UART_DISABLE(huart);
324 
325  /* Set the wake-up selection scheme */
326  MODIFY_REG(huart->Instance->CR3, USART_CR3_WUS, WakeUpSelection.WakeUpEvent);
327 
328  if (WakeUpSelection.WakeUpEvent == UART_WAKEUP_ON_ADDRESS)
329  {
330  UART_Wakeup_AddressConfig(huart, WakeUpSelection);
331  }
332 
333  /* Enable the Peripheral */
334  __HAL_UART_ENABLE(huart);
335 
336  /* Init tickstart for timeout managment*/
337  tickstart = HAL_GetTick();
338 
339  /* Wait until REACK flag is set */
340  if(UART_WaitOnFlagUntilTimeout(huart, USART_ISR_REACK, RESET, tickstart, HAL_UART_TIMEOUT_VALUE) != HAL_OK)
341  {
342  status = HAL_TIMEOUT;
343  }
344  else
345  {
346  /* Initialize the UART State */
347  huart->gState = HAL_UART_STATE_READY;
348  }

351  __HAL_UNLOCK(huart);
