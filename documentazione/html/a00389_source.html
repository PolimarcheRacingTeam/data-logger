243 void SystemCoreClockUpdate (void)
244 {
245  uint32_t tmp = 0, pllmull = 0, pllsource = 0, predivfactor = 0;
246 
247  /* Get SYSCLK source -------------------------------------------------------*/
248  tmp = RCC->CFGR & RCC_CFGR_SWS;
249 
250  switch (tmp)
251  {
252  case RCC_CFGR_SWS_HSI: /* HSI used as system clock */
253  SystemCoreClock = HSI_VALUE;
254  break;
255  case RCC_CFGR_SWS_HSE: /* HSE used as system clock */
256  SystemCoreClock = HSE_VALUE;
257  break;
258  case RCC_CFGR_SWS_PLL: /* PLL used as system clock */
259  /* Get PLL clock source and multiplication factor ----------------------*/
260  pllmull = RCC->CFGR & RCC_CFGR_PLLMUL;
261  pllsource = RCC->CFGR & RCC_CFGR_PLLSRC;
262  pllmull = ( pllmull >> 18) + 2;
263 
264 #if defined (STM32F302xE) || defined (STM32F303xE) || defined (STM32F398xx)
265  predivfactor = (RCC->CFGR2 & RCC_CFGR2_PREDIV) + 1;
266  if (pllsource == RCC_CFGR_PLLSRC_HSE_PREDIV)
267  {
268  /* HSE oscillator clock selected as PREDIV1 clock entry */
269  SystemCoreClock = (HSE_VALUE / predivfactor) * pllmull;
270  }
271  else
272  {
273  /* HSI oscillator clock selected as PREDIV1 clock entry */
274  SystemCoreClock = (HSI_VALUE / predivfactor) * pllmull;
275  }
276 #else
277  if (pllsource == RCC_CFGR_PLLSRC_HSI_DIV2)
278  {
279  /* HSI oscillator clock divided by 2 selected as PLL clock entry */
280  SystemCoreClock = (HSI_VALUE >> 1) * pllmull;
281  }
282  else
283  {
284  predivfactor = (RCC->CFGR2 & RCC_CFGR2_PREDIV) + 1;
285  /* HSE oscillator clock selected as PREDIV1 clock entry */
286  SystemCoreClock = (HSE_VALUE / predivfactor) * pllmull;
287  }
288 #endif /* STM32F302xE || STM32F303xE || STM32F398xx */
289  break;
290  default: /* HSI used as system clock */
291  SystemCoreClock = HSI_VALUE;
292  break;
293  }
294  /* Compute HCLK clock frequency ----------------*/
295  /* Get HCLK prescaler */
296  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
297  /* HCLK clock frequency */
298  SystemCoreClock >>= tmp;
299 }
